\newcommand{\workheading}[4]{

			\textbf{#1} @ \textit{#2} & \textit{#3} $|$ #4 \\
		\end{tabular*}\vspace{-4pt}
}

\newcommand{\projheading}[3]{
		\begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}

		\end{tabular*}\vspace{-4pt}
}
